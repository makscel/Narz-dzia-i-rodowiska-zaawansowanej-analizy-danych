Zbiór danych Iris

W tym rozdziale analizujemy zbiór danych Iris.

import pandas as pd
import plotly.express as px
from sklearn.datasets import load_iris

# Funkcja ładowania danych
def load_data(dataset_name):
    if dataset_name == 'iris':
        data = load_iris()
    df = pd.DataFrame(data.data, columns=data.feature_names)
    return df

# Załaduj dane Iris
df_iris = load_data('iris')

# Wyświetlenie tabeli
df_iris.head()

# Generowanie histogramów dla każdej zmiennej numerycznej
for column in df_iris.columns:
    fig = px.histogram(df_iris, x=column, nbins=20, title=f'Histogram of {column}')
    fig.show()
